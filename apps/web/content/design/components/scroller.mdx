            style={{
              fontSize: "var(--font-size-regular)",

            key={tag}
          >

          </div>
        ))}
    </div>

  </Scroller>

## Usage

```tsx
import { Scroller } from "@foundry-ui/Scroller";
```

```tsx
<Scroller>
  <List>
    {items.map((item) => (
      <ListItem key={item.key}>{item.name}</ListItem>
    ))}
  </List>
</Scroller>
```
